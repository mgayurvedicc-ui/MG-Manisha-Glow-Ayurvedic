    function orderNow(product, price){
      document.getElementById('productSelect').value = product + ' - ₹' + price;
      document.getElementById('qty').value = 1;
      location.hash = '#order';
      window.scrollTo({top: document.getElementById('order').offsetTop - 20, behavior: 'smooth'});

    function handleOrder(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const product = document.getElementById('productSelect').value;
      const qty = document.getElementById('qty').value;
      const address = document.getElementById('address').value.trim();

      if(!name || !phone){ alert('Kripya naam aur phone bharien.'); return; }

      const message = `Hello, mera naam ${name}. Main order karna chahta/chahti hoon: ${product} (Qty: ${qty}). Address: ${address}. Phone: ${phone}`;
      const waUrl = businessNumber ? `https://wa.me/${businessNumber}?text=${encodeURIComponent(message)}` : `https://wa.me/?text=${encodeURIComponent(message)}`;
      document.getElementById('orderNote').style.display = 'block';
      window.open(waUrl, '_blank');

      // Also open mail client (optional) to keep email record
      const mailto = `mailto:${contactEmail}?subject=${encodeURIComponent('New order: '+product)}&body=${encodeURIComponent(message)}`;
      window.open(mailto, '_blank');
    }

    function emailOrder(){
      const name = document.getElementById('name').value.trim();

      const qty = document.getElementById('qty').value;
